    LEDS:

      PIN 0:

        - High: Green ON
        - Low: Green OFF

        - Position: Lower Led
        - High: Red Light
        - Low: Blue Light

      PIN 14:
        - Position: Medium Led
        - High: Red Light
        - Low: Blue Light

      PIN 16:
        - Position: Upper Big Led
        - High: Red Light
        - Low: Blue Light

    BUTTONS:

      PIN 3:
        - Position: Mid Button
        - 1: Not pressed
        - 0: Pressed

      PIN 5:
        - Position: Lower Button
        - 1: Not pressed
        - 0: Pressed
  
      PIN 12:
        - Position: Upper Button
        - 1: Not pressed
        - 0: Pressed

      PIN 13: L1

        - By default, it makes the blind open/go up
